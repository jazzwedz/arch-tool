DIAGRAM 4 — ENVIRONMENT CHECKS (PowerShell)

Každý príkaz vlož samostatne do PowerShell konzoly. Zelený výstup = OK, červený = treba doinštalovať.

① Node.js povinné · min. v18 
# Skontroluj verziu Node.js
$v = node --version 2>&1
if ($LASTEXITCODE -eq 0) {
    $major = [int]($v -replace 'v(\d+).*','$1')
    if ($major -ge 18) { Write-Host "✅ Node.js $v (OK)" -ForegroundColor Green }
    else { Write-Host "❌ Node.js $v — potrebuješ v18+" -ForegroundColor Red }
} else { Write-Host "❌ Node.js nie je nainštalovaný — https://nodejs.org" -ForegroundColor Red }
② npm povinné · min. v9 
# Skontroluj verziu npm
$v = npm --version 2>&1
if ($LASTEXITCODE -eq 0) {
    $major = [int]($v -split '\.')[0]
    if ($major -ge 9) { Write-Host "✅ npm $v (OK)" -ForegroundColor Green }
    else { Write-Host "❌ npm $v — potrebuješ v9+" -ForegroundColor Red }
} else { Write-Host "❌ npm nie je nainštalovaný — príde s Node.js" -ForegroundColor Red }
③ GitHub Token povinné · repo scope 
# Nastav token a otestuj GitHub API autentifikáciu
$token = $env:GITHUB_TOKEN  # alebo vlož priamo: $token = "ghp_..."
$r = Invoke-RestMethod -Uri "https://api.github.com/user" `
     -Headers @{ Authorization="token $token"; "User-Agent"="arch-tool-check" } `
     -ErrorAction SilentlyContinue
if ($r.login) { Write-Host "✅ GitHub Token OK — prihlásený ako: $($r.login)" -ForegroundColor Green }
else { Write-Host "❌ GitHub Token neplatný alebo chýba GITHUB_TOKEN env var" -ForegroundColor Red }
④ GitHub Repo prístup povinné · read+write 
# Skontroluj prístup k dátovému repozitáru
$token = $env:GITHUB_TOKEN
$owner = $env:GITHUB_OWNER  # alebo vlož priamo: $owner = "jazzwedz"
$repo  = $env:GITHUB_REPO   # alebo vlož priamo: $repo  = "arch-data"
$r = Invoke-RestMethod -Uri "https://api.github.com/repos/$owner/$repo" `
     -Headers @{ Authorization="token $token"; "User-Agent"="arch-tool-check" } `
     -ErrorAction SilentlyContinue
if ($r.full_name) {
    $push = if ($r.permissions.push) {"✅ push"} else {"⚠️ len read"}
    Write-Host "✅ Repo $($r.full_name) dostupný — oprávnenia: $push" -ForegroundColor Green
} else { Write-Host "❌ Repo nedostupný — skontroluj GITHUB_OWNER, GITHUB_REPO, token scope" -ForegroundColor Red }
⑤ n8n Webhook voliteľné 
# Otestuj dostupnosť n8n webhook URL
$url = $env:N8N_ARCH_WEBHOOK_URL  # alebo vlož priamo: $url = "https://..."
if (-not $url) { Write-Host "⏭️  N8N_ARCH_WEBHOOK_URL nie je nastavené — doc generovanie nebude fungovať" -ForegroundColor Yellow }
else {
    try {
        $r = Invoke-WebRequest -Uri $url -Method POST `
             -Headers @{ "Content-Type"="application/json"; "X-N8N-Webhook-Secret"="arch-tool-secret-2024" } `
             -Body '{"test":true}' -TimeoutSec 10 -ErrorAction Stop
        Write-Host "✅ n8n webhook dostupný (HTTP $($r.StatusCode))" -ForegroundColor Green
    } catch {
        Write-Host "❌ n8n webhook nedostupný — $($_.Exception.Message)" -ForegroundColor Red
    }
}
⑥ Build test (voliteľné) spusti v root priečinku projektu 
# Otestuj či sa projekt dá zbuildovať (spusti v koreňovom priečinku arch-tool)
